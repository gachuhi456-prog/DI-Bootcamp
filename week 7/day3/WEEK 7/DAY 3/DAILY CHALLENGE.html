const express = require('express'); const fs = require('fs/promises'); const path = require('path'); const bcrypt = require('bcrypt'); const app = express(); const DATA_PATH = path.join(__dirname, 'users.json'); app.use(express.json()); app.use(express.urlencoded({ extended: true })); // Helper functions async function getUsers() { const data = await fs.readFile(DATA_PATH, 'utf8'); return JSON.parse(data); } async function saveUsers(users) { await fs.writeFile(DATA_PATH, JSON.stringify(users, null, 2)); } // --- ROUTES --- // POST /register app.post('/register', async (req, res) => { try { const { name, lastName, email, username, password } = req.body; const users = await getUsers(); const existingUser = users.find(u => u.username === username || u.email === email); if (existingUser) { return res.status(400).send("User already exists with that username or email."); } const hashedPassword = await bcrypt.hash(password, 10); const newUser = { id: Date.now(), name, lastName, email, username, password: hashedPassword }; users.push(newUser); await saveUsers(users); res.status(201).send(`User ${username} has been registered successfully!`); } catch (err) { res.status(500).send("Error during registration."); } }); // POST /login app.post('/login', async (req, res) => { try { const { username, password } = req.body; const users = await getUsers(); const user = users.find(u => u.username === username); if (!user) { return res.status(404).send("User not found / Not registered."); } const isMatch = await bcrypt.compare(password, user.password); if (isMatch) { res.send(`Welcome back, ${username}! Login successful.`); } else { res.status(401).send("Invalid credentials."); } } catch (err) { res.status(500).send("Server error."); } }); // GET /users - List all app.get('/users', async (req, res) => { const users = await getUsers(); res.json(users); }); // GET /users/:id app.get('/users/:id', async (req, res) => { const users = await getUsers(); const user = users.find(u => u.id == req.params.id); user ? res.json(user) : res.status(404).send("User not found"); }); // PUT /users/:id app.put('/users/:id', async (req, res) => { try { const users = await getUsers(); const index = users.findIndex(u => u.id == req.params.id); if (index === -1) return res.status(404).send("User not found"); users[index] = { ...users[index], ...req.body }; // Prevent accidental overwriting of hashed password without re-hashing logic here await saveUsers(users); res.json(users[index]); } catch (err) { res.status(500).send("Update failed."); } }); app.listen(3000, () => console.log('Server running on http://localhost:3000'));
